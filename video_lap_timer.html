// 1. 精度の高いコマ送り：fpsを手動設定 let fps = 30; const step = 1 / fps; // 2. 動画変更時にラップリセット document.getElementById("fileInput").addEventListener("change", function () { const file = this.files[0]; if (file) { video.src = URL.createObjectURL(file); video.load(); video.onloadedmetadata = () => { seekbar.max = video.duration; }; // ラップリセット laps.length = 0; lapsDiv.innerHTML = ""; } }); // 3. ラップ記録時に削除ボタン付きで表示 function recordLap() { const time = video.currentTime; const timeFixed = time.toFixed(2); const lastTime = laps.length > 0 ? laps[laps.length - 1] : 0; const diff = (time - lastTime).toFixed(2); laps.push(time); const container = document.createElement("div"); const text = document.createElement("span"); text.textContent = `Lap ${laps.length}: ${timeFixed} 秒（+${diff} 秒）`; const delBtn = document.createElement("button"); delBtn.textContent = "❌"; delBtn.style.marginLeft = "10px"; delBtn.style.fontSize = "14px"; delBtn.onclick = () => { lapsDiv.removeChild(container); // 配列から削除（完全同期を保つなら別の構造化方法も可） }; container.appendChild(text); container.appendChild(delBtn); lapsDiv.appendChild(container); }
